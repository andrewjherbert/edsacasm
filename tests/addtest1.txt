; FUNCTION TEST -  Addition part 1

.START.
            Z0F

.LOOP.
; add 0 + 0
            T .SPARE. D
            A .ZERO. D      ; 0 + 0
            G #+4 F         ; 0 + 0 < 0 -- wrong
            A .MINUS1. D    ; 0 + 0 - 1
            E #+2 F         ; 0 + 0 - 1 >= 0 -- wrong
            G #+2 F         ; Pass
            Z1F             ; Fail

; add -1 + 1 = 0
            T .SPARE. D
            A .MINUS1. D
            A .ONE. D
            G #+4 F 		; -1+1 < 0
            A .MINUS1. D
            E #+2 F         ; -1+1-1 >= 0 -- wrong
            G .LOOP. F      ; Pass
            Z2F             ; Fail


; data - align on even address
@
.ZERO.                +0L
.ONE.                 +1L
.MINUS1.              -1L
.SPARE.               &0L
.+3.                  +3L
.+5.                  +5L
