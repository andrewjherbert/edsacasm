; FUNCTION TEST - Collation part 1

.START.                Z0F

.LOOP.
; collate 0 & -1 = 0
      	       	       T .SPARE. D
		       H .ZERO. D
		       C .MINUS1. D
		       G .ERR1. F		; 0 & -1 < 0 -- wrong
		       S .ONE. D
		       G ;+2 F			; (0 & -1) - 1 < 0 -- Pass
.ERR1.                 Z1F

; collate -1 & 0 = 0
      	       	       T .SPARE. D
		       H .MINUS1. D
		       C .ZERO. D
		       G .ERR2. F		; -1 & 0 < 0 -- wrong
		       S .ONE. D
		       G ;+2 F			; (-1 & 0) - 1 < 0 -- Pass
.ERR2.                 Z2F

; collate -1 & -1 = -1
		      T .SPARE. D
		      H .MINUS1. D
		      C .MINUS1. D
		      E .ERR2. F		; -1 & -1 >= 0 -- wrong
		      A .ONE. D
		      E .LOOP. F		; -1 & -1 + 1 >= 0 -- Pass
.ERR3.		      Z3F


; data - align on even address
@
.ZERO.                +0L
.ONE.       	      +1L
.MINUS1.    	      B11111111111111111111111111111111111L
.SPARE.               +0L
