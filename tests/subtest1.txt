; FUNCTION TEST - Subtraction part 1

.START.                Z0F

.LOOP.
; subtract 0 - 0 = 0
      	       	       T .SPARE. D
		       S .ZERO. D
		       G .ERR1. F		; 0 - 0 < 0 -- Fail
		       S .ONE. D
		       G #+2 F          ; 0 - 0 - = 0 -- Pass
.ERR1.		       Z1F	 		; Fail


; subtract 1 - 1 = 0
      	       	       T .SPARE. D
		       A .ONE. D
		       S .ONE. D
		       G .ERR2. F		; 1 - 1 < 0 -- Fail
		       S .ONE. D
		       G #+2 F			; 1 - 1 = 0 -- Pass
.ERR2.		   Z2F	 		    ; Fail

; -1 - 0 = -1
		       T .SPARE. D
		       A .MINUS1. D
		       S .ZERO. D
		       E .ERR3. F		; -1 - 0 >= 0 -- wrong
		       A .ONE. D
		       E .LOOP. F		; -1 - 0 + 1 >= 0 -- Pass
.ERR3.         Z3F


; data - align on even address
@
.ZERO.                +0L
.ONE.       	      +1L
.MINUS1.              -1L
.SPARE.               +0L
