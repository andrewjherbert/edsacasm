; shift PAT1 left 1 place = PAT2
			T .SPARE. D
			A .PAT1. D
			L 0 D		; shift left 1 place
			S .PAT2. D
			G .ERR1. F      ; < 0: fail
			S .ONE. D
			G ;+2 F		; = 0: pass   

.ERR1.			Z1F

; shift PAT2 left 1 place = PAT3

			A .PAT2. D
			L 0 D
			E .ERR2. F	; >= 0: fail
			S .ONE. D

			G .START. F	; = 0: pass

.ERR2.			Z2F
