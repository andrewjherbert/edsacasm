; Test 0 -- check T order clears Acc -- should finish with 0 in Acc

.Test00.	T .DUMMY. D		; clear Acc
			Z F				; halt

; Test 1 -- check Acc ls digit -- should finish with +1 in Acc
.Test01.	T .DUMMY. D		; clear Acc
			A .PONE. D		; add +1 (0+1=1)
			Z F				; halt

; Test 2 -- check Acc ms digit -- should finish with ms bit set
.Test02.	T .DUMMY. D   	; clear Acc
			A .MONE. D		; add -1 (0+(-1)=-1)
			Z F

; Test 3 -- check odd Acc bits -- should finish with PAT1 in Acc
.Test03.	T .DUMMY. D		; clear Acc
			A .PAT1. D		; load PAT1
			Z F

; Test 4 -- check even Acc bits -- should finish with PAT2 in Acc
.Test04.	T .DUMMY. D		; clear Acc
     		A .PAT2. D      ; load PAT2
     		Z F				; halt

; Test 5 -- Arithmetic check 0 + 0 = 0 -- should finish with 0 in Acc
.Test05.	T .DUMMY. D		; clear Acc
			A .ZERO. D		; 0+0=0
			Z F				; halt

; Test 6 -- Arithmetic check (-1) + (+1) = 0 -- should finish with 0 in Acc
.Test06.	T .DUMMY. D		; clear Acc
			A .MONE. D      ; load -1
			A .PONE. D      ; add +1
			Z F				; halt

; Test 7 -- Arithmetic check -65536+1=-65535 -- should finish with all but top bit set
.Test07.	T .DUMMY. D 	; clear Acc
			A .M64K. D		; load -65536
			A .PONE. D		; add +1
			Z F				; halt

; Test 8 -- Arithmetic check 65535+1=-1 -- should finish with all but top bit set
.Test08.	T .DUMMY. D	   	; clear Acc
			A .64KM1. D   	; load 65535
			A .PONE. D		; add + 1
			Z F				; halt

; Test 9 -- Store writing check
.Test09.	T .DUMMY. D		; clear Acc
			U .STORE. D		; write to STORE -- should store 0
			Z F				; halt

; Test 10 -- Store writing check
.Test10.	T .DUMMY. D		; clear Acc
			A .PONE.	 D	; load +1
			U .STORE. D     ; store -- should store +1
			Z F				; halt

; Test 11 -- Store writing check
.Test11.   	T .DUMMY. D     ; clear Acc
			A .M64K. D		; load top bit
			U .STORE. D		; store -- should store top bit
			Z F

; Test 12 -- Store writing check
.Test12.   	T .DUMMY. D     ; clear Acc
		   	A .PAT1. D		; load PAT1
		   	U .STORE. D		; store -- should store PAT1
		   	Z F

; data align on even word boundary
@
.DUMMY.		+0L				; temporary store
.ZERO.      +0L
.PONE.		+1L
.MONE.      -1L
.64KM1.     +65535L
.M64K.      -65536L         ; top bit
.PAT1.      B10101010101010101
.STORE.     B11111111111111111  ; placed here to be easy to find using PAT1 and PAT2
.PAT2.      B01010101010101010
